📁 services/
📄 index.js

📄 main.js

导出语法 (export)

// utils.js

// 命名导出
export const PI = 3.14159;

export function add(a, b) {
  return a + b;
}

// 默认导出
export default class Calculator {
  add(a, b) { return a + b; }
}

导入语法 (import)

// main.js

// 命名导入
import { PI, add } from './utils.js';

// 默认导入
import Calculator from './utils.js';

// 导入所有
import * as Utils from './utils.js';

// 动态导入
const module = await import('./utils.js');

模块特点

1. 严格模式 ES Modules 默认使用严格模式 2. 单例模式 模块只执行一次，多次导入共享同一实例 3. 静态结构 import/export 必须在顶层 4. 异步加载 模块异步加载，不阻塞页面

在 HTML 中使用模块

<script type="module">
  import { add } from './utils.js';
  console.log(add(1, 2));
</script>
