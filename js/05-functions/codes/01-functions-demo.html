function add(a, b) { return a + b; }

// 2. 函数表达式
const subtract = function(a, b) { return a - b; };

// 3. 箭头函数
const multiply = (a, b) => a * b;
const divide = (a, b) => a / b;

  return numbers.reduce((total, n) => total + n, 0);
}
